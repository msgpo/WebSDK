watchPosition

Syntax

navigator.geolocation.watchPosition(win, fail, opts);

Arguments

winFunction

failFunction

opts (Optional)Object

Return Value

watchIdnumber

Example

var watch = function() {
  
  var win = function(p){
    alert(p.coords.latitude + " " + p.coords.longitude);
  };
  var fail = function(){
    //default query
  };
  var options = {};
  options.frequency = 3000;
  
  watchId = navigator.geolocation.watchPosition(win, fail, options);
}
